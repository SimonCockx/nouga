\begin{grammar}
<MODEL> ::= \explain{root model:}
  (<ED> | <FD>)*

<ED> ::= \explain{entity declarations:}
  `type' "D" (`extends' "E")? `:'\\
\ind <AD>*

<FD> ::= \explain{function declarations:}
  `func' "F" `:'\\
\ind `inputs' `:' <AD>*\\
\ind `output' `:' <AD>\\
\ind `assign-output' `:' <E>

<AD> ::= \explain{attribute declarations:}
  "a" <T> <CC>

<CC> ::= \explain{cardinality constraints:}
  | `(' "l" `..' "k" `)' \explain{bounded}
  | `(' "l" `..' `*' `)' \explain{unbounded}

<E> ::= \explain{expressions:}
  | <LIT>\\
  | "a"\\
  | <E> `or' <E>\\
  | <E> `and' <E>\\
  | `not' <E>\\
  | <E> (`+' | `-') <E>\\
  | <E> (`*' | `/') <E>\\
  | "D" `\{' ("a" `:' <E> (`,' "b" `:' <E>)*)? `\}'\\
  | <E> `->' "a"\\
  | <E> (`single' | `multiple')? `exists'\\
  | <E> `is' `absent'\\
  | <E> `->' "a" `only' `exists'\\
  | <E> `count'\\
  | <E> `only-element'\\
  | <E> (`all' | `any')? (`=' | `<>') <E>\\
  | <E> `contains' <E>\\
  | <E> `disjoint' <E>\\
  | `[' (<E> (`,' <E>)*)? `]'\\
  | `if' <E> `then' <E> (`else' <E>)?\\
  | "F" `(' (<E> (`,' <E>)*)? `)'\\
  | `(' <E> `)'

<LIT> ::= \explain{literals:}
  | `True' | `False' \explain{booleans}
  | "i" \explain{signed integers}
  | "r" \explain{signed decimals}
  | `empty' \explain{empty literal}

<T> ::= \explain{types:}
  | "D"\\
  | `boolean'\\
  | `int'\\
  | `number'\\
  | `nothing'
\end{grammar}
